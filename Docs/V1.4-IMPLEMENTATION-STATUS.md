// Banking API
export const banking = {
  requestLC: (data) => post('/banking/lc/request', data),
  approveLC: (lcId, data) => post('/banking/lc/approve', data),
  issueLC: (lcId, data) => post('/banking/lc/issue', data),
  getLC: (lcId) => get(`/banking/lc/${lcId}`),
  getLCsByExporter: (exporterId) => get(`/banking/lc/exporter/${exporterId}`),
  
  initiatePayment: (data) => post('/banking/payment/initiate', data),
  initiateSWIFT: (paymentId, data) => post(`/banking/payment/${paymentId}/swift/initiate`, data),
  confirmSWIFT: (paymentId, data) => post(`/banking/payment/${paymentId}/swift/confirm`, data),
  settlePayment: (paymentId, data) => post(`/banking/payment/${paymentId}/settle`, data),
};

// Forex API
export const forex = {
  requestForex: (data) => post('/forex/request', data),
  allocateForex: (data) => post('/forex/allocate', data),
  utilizeForex: (data) => post('/forex/utilize', data),
  getForex: (forexId) => get(`/forex/${forexId}`),
  getForexByExporter: (exporterId) => get(`/forex/exporter/${exporterId}`),
  
  setExchangeRate: (data) => post('/forex/rate/set', data),
  getCurrentRate: (currency) => get(`/forex/rate/${currency}`),
  getRateHistory: (currency) => get(`/forex/rate/${currency}/history`),
  
  setRetentionPolicy: (data) => post('/forex/retention/set', data),
  getRetentionPolicy: (commodityType) => get(`/forex/retention/${commodityType}`),
};

// Customs API
export const customs = {
  submitDeclaration: (data) => post('/customs/declaration/submit', data),

  clearDeclaration: (declarationId, data) => post(`/customs/declaration/${declarationId}/clear`, data),
  rejectDeclaration: (declarationId, data) => post(`/customs/declaration/${declarationId}/reject`, data),
  getDeclaration: (declarationId) => get(`/customs/declaration/${declarationId}`),
};

export const ecx = {
