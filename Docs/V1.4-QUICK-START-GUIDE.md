### **Step 2: Add Banking Functions (banking.go)**

Create `chaincodes/coffee/banking.go`:

```go
package main

import (
    "encoding/json"
    "fmt"
    "strconv"
    "time"
    "github.com/hyperledger/fabric-contract-api-go/contractapi"
)

type LetterOfCredit struct {
    LCID           string    `json:"lcId"`
    ContractID     string    `json:"contractId"`
    ExporterID     string    `json:"exporterId"`
    BankName       string    `json:"bankName"`
    Amount         float64   `json:"amount"`
    Currency       string    `json:"currency"`
    Status         string    `json:"status"`
    ExpiryDate     string    `json:"expiryDate"`
    RequestDate    time.Time `json:"requestDate"`
    ApprovalDate   string    `json:"approvalDate"`
    CreatedAt      time.Time `json:"createdAt"`
    UpdatedAt      time.Time `json:"updatedAt"`
}

func (c *CoffeeContract) RequestLC(ctx contractapi.TransactionContextInterface,
    lcID, contractID, exporterID, bankName, amountStr, currency, expiryDate string) error {
    
    amount, _ := strconv.ParseFloat(amountStr, 64)
    
    lc := LetterOfCredit{
        LCID:        lcID,
        ContractID:  contractID,
        ExporterID:  exporterID,
        BankName:    bankName,
        Amount:      amount,
        Currency:    currency,
        Status:      "REQUESTED",
        ExpiryDate:  expiryDate,
        RequestDate: time.Now(),
        CreatedAt:   time.Now(),
        UpdatedAt:   time.Now(),
    }
    
    lcJSON, _ := json.Marshal(lc)
    return ctx.GetStub().PutState("LC_"+lcID, lcJSON)
}

func (c *CoffeeContract) ApproveLC(ctx contractapi.TransactionContextInterface,
    lcID string) error {
    
    lcJSON, _ := ctx.GetStub().GetState("LC_" + lcID)
    var lc LetterOfCredit
    json.Unmarshal(lcJSON, &lc)
    
    lc.Status = "APPROVED"
    lc.ApprovalDate = time.Now().Format(time.RFC3339)
    lc.UpdatedAt = time.Now()
    
    lcJSON, _ = json.Marshal(lc)
    return ctx.GetStub().PutState("LC_"+lcID, lcJSON)
}

func (c *CoffeeContract) ReadLC(ctx contractapi.TransactionContextInterface,
    lcID string) (*LetterOfCredit, error) {
    
    lcJSON, _ := ctx.GetStub().GetState("LC_" + lcID)
    var lc LetterOfCredit
    json.Unmarshal(lcJSON, &lc)
    return &lc, nil
}

func (c *CoffeeContract) QueryLCsByExporter(ctx contractapi.TransactionContextInterface,
    exporterID string) ([]*LetterOfCredit, error) {
    
    queryString := fmt.Sprintf(`{"selector":{"exporterId":"%s"}}`, exporterID)
    resultsIterator, _ := ctx.GetStub().GetQueryResult(queryString)
    defer resultsIterator.Close()
    
    var lcs []*LetterOfCredit
    for resultsIterator.HasNext() {
        queryResponse, _ := resultsIterator.Next()
        var lc LetterOfCredit
        json.Unmarshal(queryResponse.Value, &lc)
        lcs = append(lcs, &lc)
    }
    return lcs, nil
}
```

## 🚨 **IMPORTANT NOTES**

### **Error Handling:**
Always add proper error handling:
```go
if err != nil {
    return fmt.Errorf("failed to read state: %v", err)
}
```

### **State Validation:**
Check existence before updates:
```go
if existingData == nil {
    return fmt.Errorf("record does not exist")
}
```

### **Status Transitions:**
Validate status changes:
```go
if lc.Status != "REQUESTED" {
    return fmt.Errorf("cannot approve LC in status: %s", lc.Status)
}
```

### **Timestamps:**
Always update timestamps:
```go
record.UpdatedAt = time.Now()
```
